744 edgeStruct[] sort(const edgeStruct edges[]){
745  int amount;
746  edgeStruct result[];
747  int alla[], allb[], ordera[], orderb[];
748  //split the edges into front points and rear points
749  foreach(edgeStruct ed; edges){
750  if(pointa(ed)<pointb(ed)){
751  append( alla, pointa(ed) );
752  append( allb, pointb(ed) );
753  }
754  else{
755  append( alla, pointb(ed) );
756  append( allb, pointa(ed) );
757  }
758  }
759  //Determine order
760  //First we sort by b, then by a.
761  //The order of a overrides the previous sorting, but the b order is naturally preserved for identical a values.
762  orderb = argsort(allb);
763  alla = reorder(alla, orderb);
764  allb = reorder(allb, orderb);
765 
766  ordera = argsort(alla);
767  alla = reorder(alla, ordera);
768  allb = reorder(allb, ordera);
769 
770  amount = len(alla);
771  edgeStruct temp;
772  for(int i=0; i<amount; i++){
773  temp = edgeStruct(input(edges[0]),alla[i],allb[i]);
774  push(result, temp);
775  }
776  return result;
777 }
