649 edgeStruct[] sort(const edgeStruct edges[]){
650  int amount;
651  edgeStruct result[];
652  int alla[], allb[], ordera[], orderb[];
653  //split the edges into front points and rear points
654  foreach(edgeStruct ed; edges){
655  if(pointa(ed)<pointb(ed)){
656  append( alla, pointa(ed) );
657  append( allb, pointb(ed) );
658  }
659  else{
660  append( alla, pointb(ed) );
661  append( allb, pointa(ed) );
662  }
663  }
664  //Determine order
665  //First we sort by b, then by a.
666  //The order of a overrides the previous sorting, but the b order is naturally preserved for identical a values.
667  orderb = argsort(allb);
668  alla = reorder(alla, orderb);
669  allb = reorder(allb, orderb);
670 
671  ordera = argsort(alla);
672  alla = reorder(alla, ordera);
673  allb = reorder(allb, ordera);
674 
675  amount = len(alla);
676  edgeStruct temp;
677  for(int i=0; i<amount; i++){
678  temp = edgeStruct(input(edges[0]),alla[i],allb[i]);
679  push(result, temp);
680  }
681  return result;
682 }
